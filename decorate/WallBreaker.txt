actor WallBreaker
{
Health 10
Radius 16
Height 128
Gravity 0
+SHOOTABLE
+VULNERABLE
+NOTAUTOAIMED
+NOBLOOD
+ACTIVATEMCROSS
+DONTGIB
+NOICEDEATH
DeathSound "world/barrelx"
Obituary "$OB_BARREL" // "%o went boom."
 States
  {
  	Spawn:
    	    TNT1 A 4
    	    Loop
        Death:
            FEXT B 0 Bright A_Explode(64,64)
            FEXT B 0 Bright A_SpawnDebris("ShrapnelCrash",3,3)
            FEXT B 8 Bright A_Scream
            FEXT C 6 Bright
            FEXT D 4 Bright
            Stop
	}
}

actor GasCan
{
Health 20
Radius 10
Height 48
Scale 0.8
+SOLID
+SHOOTABLE
+NOBLOOD
+ACTIVATEMCROSS
+DONTGIB
+NOICEDEATH
DeathSound "world/barrelx"
Obituary "$OB_BARREL" // "%o went boom."

    States
    {
        Spawn:
            XCAN A 4 A_JumpIf(Args[0]=0, "LoopPoint")			// if arg1 is set in doombuilder
            XCAN A 4 A_ChangeFlag("DONTTHRUST", TRUE)			// don't allow it to be pushed by explosions
		LoopPoint:
			XCAN A 4
            Loop
        Death:
            XCAN B 5 Bright A_Scream
            XCAN C 5 Bright
            FEXT C 1 Bright A_Explode//(128,192)
            FEXT C 4 Bright A_SpawnDebris("ShrapnelCrash", 5,5)
            FEXT C 5 Bright A_SetScale(1.2)
            FEXT D 5 Bright
            FEXT D 5 Bright A_SetScale(1.4)
			TNT1 A 1050 BRIGHT A_BarrelDestroy
			TNT1 A 5 A_Respawn
            Wait
     }
}
